generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

enum Role {
  ADMIN
  SHOP_OWNER
  VISITOR
}

enum OrderStatus {
  PENDING
  CONFIRMED
  SHIPPED
  DELIVERED
  CANCELLED
}
enum PaymentStatus {
  UNPAID
  PARTIAL
  PAID
}

model User {
  id        Int      @id @default(autoincrement())
  name      String
  phone     String   @unique
  password  String
  role      Role     @default(VISITOR)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  ordersAsSeller Order[] @relation("SellerOrders") // سفارشاتی که برای این فروشنده ثبت شده
  ordersAsBuyer  Order[] @relation("BuyerOrders")  // سفارشاتی که این کاربر ثبت کرده
}

model UploadBatch {
  id         Int       @id @default(autoincrement())
  filename   String
  uploadedAt DateTime  @default(now())
  products   Product[]

  @@index([uploadedAt])
}

model Product {
  id             Int         @id @default(autoincrement())
  batchId        Int
  batch          UploadBatch @relation(fields: [batchId], references: [id])
  name           String
  unitType       String
  subUnitType    String?
  countPerUnit   Int
  categoryMain   String
  categorySecond String?
  quantity       String      // original string "300+10"
  quantityMain   Int
  quantityBonus  Int
  price          Float?      // NEW – price per unit (could be nullable for older data)

  orderItems     OrderItem[]

  @@index([batchId])
}

model Order {
  id             Int          @id @default(autoincrement())
  sellerId       Int
  seller         User         @relation("SellerOrders", fields: [sellerId], references: [id])
  userId         Int
  user           User         @relation("BuyerOrders", fields: [userId], references: [id])
  notes          String?
  status         OrderStatus  @default(PENDING)
  totalAmount    Float        @default(0)   // NEW
  paidAmount     Float        @default(0)   // NEW
  paymentStatus  PaymentStatus @default(UNPAID) // NEW
  createdAt      DateTime     @default(now())
  updatedAt      DateTime     @updatedAt
  items          OrderItem[]
}

model OrderItem {
  id         Int     @id @default(autoincrement())
  orderId    Int
  order      Order   @relation(fields: [orderId], references: [id])
  productId  Int
  product    Product @relation(fields: [productId], references: [id])
  quantity   Int
  unitPrice  Float   // NEW – snapshot at order creation
  note       String?
}
